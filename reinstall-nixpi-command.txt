# Day-0 base install from a local checkout
nix run .#nixpi-deploy-ovh -- \
  --target-host root@YOUR_SERVER_IP \
  --disk /dev/disk/by-id/YOUR_TARGET_DISK_ID

# Then, on the installed machine, layer NixPI onto the host-owned /etc/nixos tree
nix run github:alexradunet/nixpi#nixpi-bootstrap-host -- \
  --primary-user alex \
  --hostname bloom-eu-1 \
  --timezone Europe/Bucharest \
  --keyboard us
